31void mapearBytesParaCodigos(node_t *tree_node, BitHuff table[], BitHuff codigo_parafolha)
32{
33 // Verifica se o nó atual é uma folha da árvore
34 // Uma folha representa um caractere que será codificado
35 if (tree_node->left == NULL && tree_node->right == NULL)
36 {
37
38 // O nó é uma folha, então armazenamos o código acumulado na tabela
39
40 // getByteFromVoidPointer(tree_node->byte) converte o byte do nó para um índice
41

43 table[recuperarByteDePonteiroVoid(tree_node)] = codigo_parafolha;

48 // Aumenta o tamanho do código acumulado para o próximo nível da árvore
49 // Isso é necessário, pois cada vez que descemos um nível na árvore, o código se torna um bit maior
50 codigo_parafolha.size++;

53 switch (1)

63 // Isso adiciona um 0 ao final do código, representando a escolha do caminho esquerdo
64 leftCode.bitH <<= 1;
65 // Chama recursivamente a função para processar o nó esquerdo
66 mapearBytesParaCodigos(tree_node->left, table, leftCode);
67 }

69
70 // Caso 2: Processa o filho direito do nó atual
71 case 2:
72 // Verifica se o nó direito não é nulo
73 if (tree_node->right != NULL)
74 {
75 // Cria uma cópia do código acumulado para a chamada recursiva do filho direito
76 BitHuff rightCode = codigo_parafolha;
77 // Desloca o código para a esquerda e adiciona um 1 ao final do código
78 // Isso representa a escolha do caminho direito

89int tamanho_lixo(int frequency[], BitHuff table[])
90{
91 long bits_antes = 0;
92 long bits_depois = 0;
93
94 int i = 0;
95 while (i < 256)
96 {
97 if (frequency[i] != 0)
98 {
99 // bits_antes é a quantidade de bits original (sem compressao)
100 bits_antes += frequency[i] * 8;
101 // multiplicado por 8 pq i eh armazenado em um byte antes da compressao
102
103 // bits_depois é a quantidade total de bits após a compressão
104 // Usa o tamanho do código em huff armazenado em table[i].size
105 bits_depois += frequency[i] * table[i].size;
106 }
107 i++;
108 }
109
110 // Calcula o "lixo" (bits não usados no último byte)
111
112 // diferença entre bits_antes e bits_depois indica quantos bits foram "perdidos" ou não utilizados.
113
114 int trash = (bits_antes - bits_depois) % 8;
115
116 return trash;
117}

124void escrever_no_Cabecalho_Arquivo_compac(FILE *file, int trashSize, int treeSize)
125{
126 unsigned char primeiro_byte = 0x00; // Inicializa o primeiro byte com 0
127 unsigned char segundo_byte = 0x00; // Inicializa o segundo byte com 0
128
129 // Primeiro byte:
130 // - Pega os 3 bits mais significativos do tamanho do lixo (trashSize)
131 // - Usa o operador << para deslocar o trashSize 5 bits para a esquerda, colocando-o nos 3 bits mais significativos do byte.
132 primeiro_byte = primeiro_byte | (trashSize << 5);
133
134 // - Agora, pega os 5 bits mais significativos do tamanho da árvore (treeSize)
135 // - Usa o operador >> para deslocar 8 bits para a direita, movendo os 5 bits mais significativos para o lugar correto.
136 primeiro_byte = primeiro_byte | (treeSize >> 8);
137
138 // Segundo byte:
139 // - Aqui colocamos os 8 bits menos significativos do tamanho da árvore
140 // - Como os 8 bits menos significativos já estão no lugar certo, basta fazer uma operação OR simples.
141 segundo_byte = segundo_byte | treeSize;
142
143 // Escreve os dois bytes no arquivo
144 fwrite(&primeiro_byte, sizeof(unsigned char), 1, file); // Escreve o primeiro byte
145 fwrite(&segundo_byte, sizeof(unsigned char), 1, file); // Escreve o segundo byte
146}

192// Essa função lê o arquivo de entrada fileIn byte por byte,
193// converte cada byte em seu código correspondente de Huffman (usando a tabela BitHuff)
194// e grava esses códigos compactados no arquivo de saída fileOut
195
196void gravarCodigos(FILE *fileIn, FILE *fileOut, BitHuff table[], int trashSize)
197{
198 uint8_t acumulador_bits = 0; // Acumulador para armazenar os bits temporários (8 bits), que serão escritos no arquivo de saída.
199 int tamanho_acumulador = 0; // Contador para a quantidade de bits armazenados no acumulador, para saber quando ele está "cheio".
200 int caractere; // Variável para armazenar temporariamente cada caractere lido do arquivo de entrada.
201
202 // Lê o arquivo de entrada byte por byte
203 // A função 'getc' lê um caractere (byte) por vez do arquivo 'fileIn'. A leitura continua até atingir o final do arquivo (EOF).
204 while ((caractere = getc(fileIn)) != EOF)
205 {
206 BitHuff entry = table[caractere]; // A tabela de Huffman ('table') contém o código binário (compressão) para cada caractere.
207
208 // Aqui, 'entry' armazena o código binário e seu tamanho (em bits) correspondente ao caractere lido.
209
210 // Percorre cada bit do código de Huffman correspondente ao caractere atual
211 // Esse loop percorre o código de Huffman do último bit até o primeiro (do bit mais significativo ao menos significativo).
212 for (int i = entry.size - 1; i >= 0; --i)
213 {
214 // Inserir o próximo bit no acumulador
215 // A operação 'acumulador_bits << 1' desloca os bits do acumulador uma posição para a esquerda, abrindo espaço para o novo bit.
216 // '((entry.bitH >> i) & 1)' extrai o i-ésimo bit do código Huffman (1 ou 0) e o adiciona no acumulador.
217 acumulador_bits = (acumulador_bits << 1) | ((entry.bitH >> i) & 1);
218 tamanho_acumulador++; // Incrementa o contador, indicando que mais um bit foi adicionado ao acumulador.
219
220 // Quando o acumulador estiver cheio (8 bits completos), ele deve ser escrito no arquivo de saída.
221 if (tamanho_acumulador == 8)
222 {
223 fwrite(&acumulador_bits, sizeof(uint8_t), 1, fileOut); // Escreve 1 byte (8 bits) no arquivo de saída 'fileOut'.
224 acumulador_bits = 0; // Reseta o acumulador para que ele possa armazenar novos bits.
225 tamanho_acumulador = 0; // Reinicia o contador para os próximos bits a serem acumulados.
226 }
227 }
228 }
229
230 // Após ler todos os bytes do arquivo de entrada, pode haver bits restantes no acumulador que não foram escritos ainda.
231 // Isso acontece quando o arquivo termina mas o acumulador não chegou a 8 bits completos.
232 if (tamanho_acumulador > 0)
233 {
234 // Adiciona o "lixo" (bits extras que não fazem parte dos dados) ao final do último byte.
235 // O 'trashSize' indica quantos desses bits finais são irrelevantes para a informação, necessários para completar o último byte.
236 acumulador_bits <<= trashSize;
237
238 // Escreve o último byte (com os bits restantes e os bits de "lixo") no arquivo de saída.
239 fwrite(&acumulador_bits, sizeof(uint8_t), 1, fileOut);
240 }
241}
